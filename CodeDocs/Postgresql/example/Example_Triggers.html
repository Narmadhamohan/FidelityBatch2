INSERT Log new employee creation in employee_creation_logs 

CREATE OR REPLACE FUNCTION log_new_employee()
RETURNS TRIGGER AS $$
BEGIN
    INSERT INTO employee_creation_logs (employee_id, name, salary, creation_date)
    VALUES (NEW.employee_id, NEW.name, NEW.salary, NOW());
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER log_new_employee_trigger
AFTER INSERT ON employees
FOR EACH ROW EXECUTE FUNCTION log_new_employee();

UPDATE Log salary changes in salary_changes 

CREATE OR REPLACE FUNCTION log_salary_change()
RETURNS TRIGGER AS $$
BEGIN
    INSERT INTO salary_changes (employee_id, old_salary, new_salary) 
    VALUES (NEW.employee_id, OLD.salary, NEW.salary);
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER log_salary_change_trigger
AFTER UPDATE OF salary ON employees
FOR EACH ROW EXECUTE FUNCTION log_salary_change();
salary_changes:
Log Created
  • employee_id: 101
  • old_salary: 50000
  • new_salary: 60000
DELETE Log employee deletion in salary_changes 

CREATE OR REPLACE FUNCTION log_employee_deletion()
RETURNS TRIGGER AS $$
BEGIN
    INSERT INTO salary_changes (employee_id, old_salary, new_salary) 
    VALUES (OLD.employee_id, OLD.salary, NULL);
    RETURN OLD;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER log_employee_deletion_trigger
AFTER DELETE ON employees
FOR EACH ROW EXECUTE FUNCTION log_employee_deletion();
salary_changes:
Log Created
  • employee_id: 103
  • old_salary: 75000
  • new_salary: NULL
